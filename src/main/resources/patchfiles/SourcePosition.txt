package de.hhu.stups.sablecc.patch;

public class SourcePosition {

	private final int line;

	private final int pos;

	public SourcePosition(final int line, final int pos) {
		this.line = line;
		this.pos = pos;
	}

	public int getLine() {
		return line;
	}

	public int getPos() {
		return pos;
	}
}
